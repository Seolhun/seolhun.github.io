## 3. Code
```tsx
function countingSort(arr) {
    return arr.sort(function(a, b) {
        if(a < b) {
            return -1;
        }
        if(a > b) {
            return 1;
        }
        return 0;
    });
}

function main() {
    var n = parseInt(readLine());
    arr = readLine().split(' ');
    arr = arr.map(Number);
    var result = countingSort(arr);
    console.log(result.join(' '))
}
```
